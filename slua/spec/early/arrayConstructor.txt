New Syntax {
	
	exp @::= | "[" exp ";" comptimeExp "]"
}

Array Constructor {
	
	Construct an array of one item X many times
	
	[x;y] -> x the element, y how many of it.
	x must be copyable (any kind).
	y must be a non-negative integer.
	if y is 0, then x will not be evaluated.
	
}
