New Syntax {

	ExportPrefix ::= "ex"
	
	OptExportPrefix ::= [ ExportPrefix ]
}

New Keywords {
	ex
}

Export Prefix {
	
	Exports a type, function, variable, etc, for use inside other files, projects, modules, ...
	
	"ex" is easy to type, isnt rly used.
	
	"pub" -> too hard to type
	"pb" -> hard to understand
	"exp" -> exponent
	"expo" -> exponent, looks odd
	"export" -> too long, and also kinda common
	"ext" -> too common "cuz extension"
	"exs" -> hard to type
	
}
